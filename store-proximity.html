📋 SQL Setup (run once in Supabase)

Paste this in your Supabase SQL Editor:

CREATE TABLE shopping_sessions (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id uuid REFERENCES auth.users(id),
  session_date timestamptz DEFAULT now(),
  total_amount numeric(10,2) DEFAULT 0,
  store_count int DEFAULT 0
);

CREATE TABLE purchases (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  session_id uuid REFERENCES shopping_sessions(id),
  user_id uuid REFERENCES auth.users(id),
  store_id text, store_name text,
  store_address text, product_id int,
  product_name text, quantity numeric(8,3),
  unit_price numeric(8,2),
  line_total numeric(10,2),
  purchased_at timestamptz DEFAULT now()
);

ALTER TABLE shopping_sessions ENABLE ROW LEVEL SECURITY;
ALTER TABLE purchases ENABLE ROW LEVEL SECURITY;
CREATE POLICY "own" ON shopping_sessions FOR ALL USING (auth.uid()=user_id);
CREATE POLICY "own" ON purchases FOR ALL USING (auth.uid()=user_id);
